/** * ============================================ * TEACHER PANEL - QOMed Platform * ============================================ * Dashboard completo con: * - Gestión de clases * - Actividad de estudiantes * - Registro de accesos * - Lista de estudiantes */ class TeacherPanel { constructor() { this.session = getCurrentSession(); if (!this.session || (this.session.rol !== 'profesor' && this.session.rol !== 'admin')) { this.redirectUnauthorized(); return; } this.charts = {}; this.init(); } redirectUnauthorized() { if (this.session) { const redirectMap = { 'alumno': 'student.html', 'admin': 'admin.html' }; window.location.href = redirectMap[this.session.rol] || 'index.html'; } else { window.location.href = 'index.html'; } } init() { this.setupNavigation(); this.loadClasses(); this.loadStats(); this.setupEventListeners(); this.populateFilters(); } // ============================================ // NAVEGACIÓN // ============================================ setupNavigation() { document.querySelectorAll('.nav-item').forEach(item => { item.addEventListener('click', () => { const section = item.dataset.section; // Actualizar menú document.querySelectorAll('.nav-item').forEach(ni => ni.classList.remove('active')); item.classList.add('active'); // Mostrar sección document.querySelectorAll('.content-section').forEach(s => s.classList.remove('active')); const sectionEl = document.getElementById(`${section}Section`); if (sectionEl) { sectionEl.classList.add('active'); // Cargar datos según sección if (section === 'activity') this.loadStudentActivity(); if (section === 'access') this.loadAccessLog(); if (section === 'students') this.loadStudentsList(); } }); }); } // ============================================ // FILTROS // ============================================ populateFilters() { const myClasses = this.getMyClasses(); // Poblar todos los select de clase ['activityClassFilter', 'accessClassFilter', 'studentClassFilter'].forEach(filterId => { const select = document.getElementById(filterId); if (select) { select.innerHTML = ''; myClasses.forEach(cls => { select.innerHTML += ``; }); } }); // Poblar filtro de estudiantes const allEnrollments = this.getAllEnrollments(); const studentIds = [...new Set(allEnrollments.map(e => e.studentId))]; const accessStudentFilter = document.getElementById('accessStudentFilter'); if (accessStudentFilter) { accessStudentFilter.innerHTML = ''; studentIds.forEach(id => { const student = db.getUserById(id); if (student) { accessStudentFilter.innerHTML += ``; } }); } } // ============================================ // SECCIÓN 1: CLASES // ============================================ getMyClasses() { const allClasses = db.getClasses(); const userId = this.session.id || this.session.userId; return this.session.rol === 'admin' ? allClasses : allClasses.filter(c => c.teacherId === userId); } getAllEnrollments() { const myClasses = this.getMyClasses(); const myClassIds = myClasses.map(c => c.id); const allEnrollments = db.getEnrollments(); return allEnrollments.filter(e => myClassIds.includes(e.classId)); } getMyStudents() { const enrollments = this.getAllEnrollments(); const studentIds = [...new Set(enrollments.map(e => e.studentId))]; return studentIds.map(id => db.getUserById(id)).filter(Boolean); } loadStats() { const classes = this.getMyClasses(); const enrollments = this.getAllEnrollments(); let totalStudents = enrollments.length; let totalProgress = 0; let completedCount = 0; enrollments.forEach(enr => { totalProgress += enr.progress || 0; if (enr.completed) completedCount++; }); const avgProgress = totalStudents > 0 ? Math.round(totalProgress / totalStudents) : 0; document.getElementById('totalClasses').textContent = classes.length; document.getElementById('totalStudents').textContent = totalStudents; document.getElementById('avgProgress').textContent = avgProgress + '%'; document.getElementById('completedStudents').textContent = completedCount; } loadClasses() { const classes = this.getMyClasses(); const grid = document.getElementById('classesGrid'); if (!grid) return; grid.innerHTML = ''; if (classes.length === 0) { grid.innerHTML = '

No hay clases registradas. ¡Crea tu primera clase!

'; return; } classes.forEach(classItem => { grid.appendChild(this.createClassCard(classItem)); }); } createClassCard(classItem) { const enrollments = db.getClassProgress(classItem.id); const studentsCount = enrollments.length; const avgProgress = utils.calculateAverageProgress(enrollments); const completedCount = enrollments.filter(e => e.completed).length; const div = document.createElement('div'); div.className = 'class-card'; div.style.borderTopColor = classItem.color || '#FFD700'; div.innerHTML = `

${classItem.title}

${utils.truncate(classItem.description, 100)}

👥 ${studentsCount} alumnos ✅ ${completedCount} completados
📊 ${avgProgress}% promedio
`; return div; } // ============================================ // SECCIÓN 2: ACTIVIDAD DE ESTUDIANTES // ============================================ loadStudentActivity() { const classFilter = document.getElementById('activityClassFilter')?.value; const enrollments = this.getAllEnrollments(); // Filtrar por clase si es necesario const filteredEnrollments = classFilter ? enrollments.filter(e => e.classId === classFilter) : enrollments; // Tabla de actividad const tbody = document.getElementById('activityTableBody'); if (!tbody) return; tbody.innerHTML = ''; if (filteredEnrollments.length === 0) { tbody.innerHTML = 'No hay actividad registrada'; } else { filteredEnrollments.forEach(enr => { const student = db.getUserById(enr.studentId); const classItem = db.getClassById(enr.classId); if (!student) return; // Contar accesos para este estudiante en esta clase const activities = db.getActivitiesByUser(enr.studentId); const classAccesses = activities.filter(a => a.page.includes(enr.classId) || a.page === `access_class_${enr.classId}` ).length; const statusBadge = enr.completed ? '✅ Completado' : enr.progress > 0 ? '🔄 En Progreso' : '📋 Inscrito'; const tr = document.createElement('tr'); tr.innerHTML = ` ${student.nombre} ${classItem ? classItem.title : 'N/A'}
${enr.progress}% ${utils.formatRelativeTime(enr.lastAccess)} ${classAccesses} ${statusBadge} `; tbody.appendChild(tr); }); } // Cargar gráficos this.loadActivityCharts(filteredEnrollments); } loadActivityCharts(enrollments) { // Destruir gráficos anteriores Object.values(this.charts).forEach(chart => chart.destroy()); this.charts = {}; // Gráfico de accesos por día this.loadAccessChart(); // Gráfico de distribución de progreso this.loadProgressDistributionChart(enrollments); } loadAccessChart() { const ctx = document.getElementById('accessChart'); if (!ctx) return; const last7Days = this.getLast7DaysAccesses(); this.charts.access = new Chart(ctx, { type: 'bar', data: { labels: last7Days.labels, datasets: [{ label: 'Accesos', data: last7Days.data, backgroundColor: 'rgba(255, 215, 0, 0.6)', borderColor: '#FFD700', borderWidth: 1, borderRadius: 5 }] }, options: { responsive: true, plugins: { legend: { display: false } }, scales: { y: { beginAtZero: true, ticks: { stepSize: 1 } } } } }); } loadProgressDistributionChart(enrollments) { const ctx = document.getElementById('progressChart'); if (!ctx) return; // Agrupar por rangos de progreso const ranges = { '0%': 0, '1-25%': 0, '26-50%': 0, '51-75%': 0, '76-99%': 0, '100%': 0 }; enrollments.forEach(enr => { const p = enr.progress || 0; if (p === 0) ranges['0%']++; else if (p <= 25) ranges['1-25%']++; else if (p <= 50) ranges['26-50%']++; else if (p <= 75) ranges['51-75%']++; else if (p < 100) ranges['76-99%']++; else ranges['100%']++; }); this.charts.progress = new Chart(ctx, { type: 'doughnut', data: { labels: Object.keys(ranges), datasets: [{ data: Object.values(ranges), backgroundColor: [ '#e74c3c', '#f39c12', '#f1c40f', '#2ecc71', '#27ae60', '#1abc9c' ] }] }, options: { responsive: true, plugins: { legend: { position: 'bottom' } } } }); } getLast7DaysAccesses() { const myClasses = this.getMyClasses(); const myClassIds = myClasses.map(c => c.id); const activities = db.getActivities(); const labels = []; const data = []; for (let i = 6; i >= 0; i--) { const date = new Date(); date.setDate(date.getDate() - i); const dateStr = date.toISOString().split('T')[0]; labels.push(date.toLocaleDateString('es-ES', { weekday: 'short' })); const count = activities.filter(a => { const aDate = new Date(a.timestamp).toISOString().split('T')[0]; return aDate === dateStr && myClassIds.some(cId => a.page.includes(cId)); }).length; data.push(count); } return { labels, data }; } // ============================================ // SECCIÓN 3: REGISTRO DE ACCESOS // ============================================ loadAccessLog() { const studentFilter = document.getElementById('accessStudentFilter')?.value; const classFilter = document.getElementById('accessClassFilter')?.value; const myClasses = this.getMyClasses(); const myClassIds = myClasses.map(c => c.id); let activities = db.getActivities() .filter(a => myClassIds.some(cId => a.page.includes(cId))) .sort((a, b) => new Date(b.timestamp) - new Date(a.timestamp)); // Aplicar filtros if (studentFilter) { activities = activities.filter(a => a.userId === studentFilter); } if (classFilter) { activities = activities.filter(a => a.page.includes(classFilter)); } // Actualizar contadores const today = new Date().toISOString().split('T')[0]; const todayCount = activities.filter(a => a.timestamp.startsWith(today)).length; const activeStudents7Days = [...new Set( activities.filter(a => { const aDate = new Date(a.timestamp); const sevenDaysAgo = new Date(); sevenDaysAgo.setDate(sevenDaysAgo.getDate() - 7); return aDate >= sevenDaysAgo; }).map(a => a.userId) )].length; const totalSessionTime = activities.reduce((sum, a) => sum + (a.sessionTime || 0), 0); const avgSessionTime = activities.length > 0 ? Math.round(totalSessionTime / activities.length) : 0; document.getElementById('totalAccesses').textContent = activities.length; document.getElementById('todayAccesses').textContent = todayCount; document.getElementById('activeStudentsCount').textContent = activeStudents7Days; document.getElementById('avgSessionTime').textContent = avgSessionTime + ' min'; // Tabla const tbody = document.getElementById('accessTableBody'); if (!tbody) return; tbody.innerHTML = ''; if (activities.length === 0) { tbody.innerHTML = 'No hay accesos registrados'; } else { activities.slice(0, 50).forEach(activity => { const student = db.getUserById(activity.userId); const classItem = myClasses.find(c => activity.page.includes(c.id)); let action = 'Acceso'; if (activity.page.includes('enrolled')) action = '📝 Inscripción'; else if (activity.page.includes('access')) action = '📚 Accedió a clase'; else if (activity.page === 'login') action = '🔑 Inicio de sesión'; const tr = document.createElement('tr'); tr.innerHTML = ` ${utils.formatDate(activity.timestamp)} ${student ? student.nombre : 'Desconocido'} ${classItem ? classItem.title : 'N/A'} ${action} ${activity.sessionTime || 0}s `; tbody.appendChild(tr); }); } } // ============================================ // SECCIÓN 4: LISTA DE ESTUDIANTES // ============================================ loadStudentsList() { const classFilter = document.getElementById('studentClassFilter')?.value; const searchTerm = document.getElementById('studentSearch')?.value?.toLowerCase() || ''; const myStudents = this.getMyStudents(); const enrollments = this.getAllEnrollments(); // Filtrar por clase let filteredStudents = myStudents; if (classFilter) { const classEnrollments = enrollments.filter(e => e.classId === classFilter); const studentIds = classEnrollments.map(e => e.studentId); filteredStudents = myStudents.filter(s => studentIds.includes(s.id)); } // Filtrar por búsqueda if (searchTerm) { filteredStudents = filteredStudents.filter(s => s.nombre.toLowerCase().includes(searchTerm) || s.email.toLowerCase().includes(searchTerm) ); } // Actualizar contadores let totalProgress = 0; let activeCount = 0; let completedCount = 0; filteredStudents.forEach(student => { const studentEnrollments = enrollments.filter(e => e.studentId === student.id); const avg = utils.calculateAverageProgress(studentEnrollments); totalProgress += avg; if (studentEnrollments.some(e => e.completed)) completedCount++; if (studentEnrollments.some(e => e.progress > 0 && !e.completed)) activeCount++; }); const globalAvg = filteredStudents.length > 0 ? Math.round(totalProgress / filteredStudents.length) : 0; document.getElementById('totalEnrolledStudents').textContent = filteredStudents.length; document.getElementById('activeStudents').textContent = activeCount; document.getElementById('completedStudentsCount').textContent = completedCount; document.getElementById('globalAvgProgress').textContent = globalAvg + '%'; // Tabla const tbody = document.getElementById('studentsTableBody'); if (!tbody) return; tbody.innerHTML = ''; if (filteredStudents.length === 0) { tbody.innerHTML = 'No se encontraron estudiantes'; } else { filteredStudents.forEach(student => { const studentEnrollments = enrollments.filter(e => e.studentId === student.id); const avgProgress = utils.calculateAverageProgress(studentEnrollments); const classesCount = studentEnrollments.length; const lastAccess = studentEnrollments .map(e => e.lastAccess) .sort((a, b) => new Date(b) - new Date(a))[0]; const hasCompleted = studentEnrollments.some(e => e.completed); const hasActive = studentEnrollments.some(e => e.progress > 0 && !e.completed); const statusBadge = hasCompleted ? '✅ Completado' : hasActive ? '🔄 En Progreso' : '📋 Sin actividad'; const tr = document.createElement('tr'); tr.innerHTML = ` ${student.nombre} ${student.email} ${classesCount} clase(s)
${avgProgress}% ${lastAccess ? utils.formatRelativeTime(lastAccess) : 'Nunca'} ${statusBadge} `; tbody.appendChild(tr); }); } } viewStudentDetail(studentId) { const student = db.getUserById(studentId); if (!student) return; const enrollments = this.getAllEnrollments().filter(e => e.studentId === studentId); // Mostrar en modal de progreso con información de este estudiante let modalTitle = `Actividad de: ${student.nombre}`; if (enrollments.length > 0) { const classItem = db.getClassById(enrollments[0].classId); document.getElementById('progressClassName').textContent = classItem ? `${modalTitle} - ${classItem.title}` : modalTitle; } else { document.getElementById('progressClassName').textContent = modalTitle; } document.getElementById('modalTotalStudents').textContent = '1'; document.getElementById('modalAvgProgress').textContent = utils.calculateAverageProgress(enrollments) + '%'; document.getElementById('modalCompleted').textContent = enrollments.filter(e => e.completed).length; const tbody = document.getElementById('progressTableBody'); tbody.innerHTML = ''; enrollments.forEach(enr => { const classItem = db.getClassById(enr.classId); const tr = document.createElement('tr'); tr.innerHTML = ` ${classItem ? classItem.title : 'N/A'} ${student.email}
${enr.progress}% ${utils.formatRelativeTime(enr.lastAccess)} Acceder `; tbody.appendChild(tr); }); document.getElementById('progressModal').classList.add('active'); } // ============================================ // GESTIÓN DE CLASES (EXISTENTE) // ============================================ openClassModal(classId = null) { const modal = document.getElementById('classModal'); const form = document.getElementById('classForm'); if (!modal || !form) return; form.reset(); if (classId) { const classItem = db.getClassById(classId); if (classItem) { document.getElementById('classId').value = classItem.id; document.getElementById('classTitle').value = classItem.title; document.getElementById('classDescription').value = classItem.description; document.getElementById('classLink').value = classItem.link; document.getElementById('classColor').value = classItem.color || '#4CAF50'; document.getElementById('classModalTitle').textContent = 'Editar Clase'; } } else { document.getElementById('classId').value = ''; document.getElementById('classColor').value = utils.getRandomColor(); document.getElementById('classModalTitle').textContent = 'Nueva Clase'; } modal.classList.add('active'); } editClass(classId) { this.openClassModal(classId); } deleteClass(classId) { if (!utils.confirm('¿Eliminar esta clase y todos sus datos?')) return; db.deleteClass(classId); utils.showToast('Clase eliminada', 'success'); this.loadClasses(); this.loadStats(); } viewProgress(classId) { const classItem = db.getClassById(classId); if (!classItem) return; document.getElementById('progressClassName').textContent = classItem.title; const enrollments = db.getClassProgress(classId); document.getElementById('modalTotalStudents').textContent = enrollments.length; document.getElementById('modalAvgProgress').textContent = utils.calculateAverageProgress(enrollments) + '%'; document.getElementById('modalCompleted').textContent = enrollments.filter(e => e.completed).length; const tbody = document.getElementById('progressTableBody'); tbody.innerHTML = ''; if (enrollments.length === 0) { tbody.innerHTML = 'No hay alumnos inscritos'; } else { enrollments.forEach(enrollment => { const student = db.getUserById(enrollment.studentId); if (student) { const tr = document.createElement('tr'); tr.innerHTML = ` ${student.nombre} ${student.email}
${enrollment.progress}% ${utils.formatRelativeTime(enrollment.lastAccess)} Acceder `; tbody.appendChild(tr); } }); } document.getElementById('progressModal').classList.add('active'); } closeProgressModal() { document.getElementById('progressModal')?.classList.remove('active'); } closeModal() { document.getElementById('classModal')?.classList.remove('active'); } saveClass() { const classId = document.getElementById('classId').value; const userId = this.session.id || this.session.userId; const classData = { title: document.getElementById('classTitle').value, description: document.getElementById('classDescription').value, link: document.getElementById('classLink').value, color: document.getElementById('classColor').value, teacherId: userId }; try { if (classId) { db.updateClass(classId, classData); utils.showToast('Clase actualizada', 'success'); } else { db.createClass(classData); utils.showToast('Clase creada', 'success'); } this.closeModal(); this.loadClasses(); this.loadStats(); this.populateFilters(); } catch (error) { utils.showToast('Error al guardar', 'error'); } } setupEventListeners() { const classForm = document.getElementById('classForm'); if (classForm) { classForm.addEventListener('submit', (e) => { e.preventDefault(); this.saveClass(); }); } window.addEventListener('click', (e) => { if (e.target.classList.contains('modal')) { e.target.classList.remove('active'); } }); } } // Inicialización SEGURA if (window.location.pathname.includes('teacher.html') || window.location.pathname.includes('admin.html')) { window.addEventListener('load', () => { if (!window.teacherPanel) { window.teacherPanel = new TeacherPanel(); } }); }
